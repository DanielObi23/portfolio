export function makeScreenshots(
  project: string,
  count: number,
) {
  return Array.from({ length: count }, (_, i) => {
    const n = i + 1;
    return {
      id: `${project}-screenshot-${n}`,
      title: `screenshot-${n}`,
      fileType: "img" as const,
      imageUrl: `/screenshots/${project}/${project}-screenshot-${n}.jpg`,
    };
  });
}
